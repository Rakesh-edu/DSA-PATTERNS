->Important patterns:
    1.Find the given element k /index i  in a sorted array
    ex:
        while(low<=high){
            int mid=low+(high-low)/2; //to handle integer overflow edgecase
            if(arr[mid]==k){
                return ans;
            }
            else if(arr[mid]<k){
                low=mid+1;
            }
            else{
                high = mid-1;
            }
        }

        2.Finding on UPPER BOUND/LOWER BOUND
            while(low<=high){
            int mid=low+(high-low)/2; //to handle integer overflow edgecase
            if(arr[mid]==k){
                return ans;
            }
            else if(arr[mid]<k){
                ans=mid;            //for lower bound
                low=mid+1;
            }
            else{
                ans=mid             //for upper bound
                high = mid-1;
            }
        }
