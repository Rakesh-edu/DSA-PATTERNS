                        class Solution {
                        public:
                            int countBinarySubstrings(string s) {
                                int ans=0;
                                for(int i=0;i<s.size();i++){
                                    int count=0;
                                    for(int j=i;j<s.size();j++){
                                        if(s[j]=='0')count++;
                                        else count--;

                                        if(count==0){
                                            ans++;break;
                                        }
                                    }
                                    
                                }
                                return ans;
                            }
                        };
